    class Letter{
        char c;
    }
    void f(Letter y){
        y.c='z';
        System.out.println(y.c);
    }
    public static void main(String[] args) {
        Letter x=new Letter();
        x.c='a';
        System.out.println(x.c);
        f(x);
        System.out.println(x.c);
    }
    //output
    a
    z
    z
